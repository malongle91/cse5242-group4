CREATE TABLE Student (
    student_id INTEGER PRIMARY KEY,
    first_name text NOT NULL, 
    last_name text NOT NULL, 
    seat_group_no INTEGER NOT NULL, 
    num_answered INTEGER, 
    num_correct INTEGER, 
    is_submitted FALSE
);

CREATE TABLE Question ( 
    question_id INTEGER PRIMARY KEY, 
    correct_answer CHAR NOT NULL
);

CREATE TABLE Student_Answer (
    question_id INTEGER, 
    student_id INTEGER, 
    student_answer CHAR,
    FOREIGN KEY(question_id) REFERENCES Question(question_id),
    FOREIGN KEY(student_id) REFERENCES Student(student_id)
);

CREATE TABLE Hotspot (
    seat_group_no INTEGER NOT NULL,
    student_id INTEGER NOT NULL,
    incorrect_answer CHAR NOT NULL,
    PRIMARY KEY(seat_group_no, student_id),
    FOREIGN KEY(seat_group_no) REFERENCES Student(seat_group_no),
    FOREIGN KEY(student_id) REFERENCES Student(student_id)
);
